hmacfs/                       # 프로젝트 루트
├── src/                      # 소스 코드 디렉토리

│   ├── fuse_operations.h
│   ├── hmac.c                # HMAC 계산/검증 모듈 (OpenSSL 등 활용)
│   ├── hmac.h
│   ├── integrity.c           # 무결성 검증 로직 (HMAC 비교, 결과 처리)
│   ├── integrity.h
│   ├── log.c                 # 로그 및 알림 처리 모듈
│   ├── log.h
│   └── utils.c               # 기타 유틸 함수 (경로 처리 등)
│       utils.h
│
├── include/                  # 공용 헤더 파일
│   ├── config.h              # 전역 상수, 설정값 정의 (HMAC 키 파일 경로, 로그 경로 등)
│   └── common.h              # 공통 매크로, 에러 처리 등
│
├── tests/                    # 단위 테스트 코드
│   ├── test_hmac.c           # HMAC 모듈 테스트
│   ├── test_integrity.c      # 무결성 검증 테스트
│   └── test_fs.c             # FUSE 연산 테스트 (mock 활용)
│
├── build/                    # 빌드 아티팩트 (object, binary)
│
├── scripts/                  # 유틸리티 스크립트
│   ├── run.sh                # 파일시스템 실행 스크립트 (mount/unmount)
│   └── keygen.sh             # HMAC 키 생성 스크립트 (openssl rand 등 활용)
│
├── docs/                     # 설계 문서 및 사용 설명서
│   ├── design.md             # 파일시스템 설계 문서
│   ├── usage.md              # 실행 및 사용 방법
│   └── threat_model.md       # 위협 모델링 및 보안 고려사항
│
├── Makefile                  # 빌드 스크립트
├── README.md                 # 프로젝트 개요
└── .gitignore
